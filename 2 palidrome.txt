	    for i in range(len(list1)):

	    else:
	        return False

def revstr(string):
	    revstr=""
	    for i in range(len(string)):
	        revstr=string[i]+revstr
	    return revstr
	

def str_count(str1, char):
	   count=0
	   for i in range(len(str1)):
	       if char==str1[i]:
	           count+=1
	   return count

def is_palindrome(str):
	
	    for i in range(0, int(len(str)/2)):
	
	        if str[i] != str[len(str)-i-1]:
	
	            return False
	
	    return True
