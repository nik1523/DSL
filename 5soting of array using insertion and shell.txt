        value=list1[i]
        j=i-1
        while j >= 0 and value < list1[j]:

            j -= 1  
            list1[j + 1] = value

def shellsort(array,n):
    interval=n//2
    while interval >=0:
        for i in range(interval,n):
            temp=array[i]
            j=i
            while j>=interval and array[j-interval]>=temp:
                array[j]=array[j-interval]
                j-=interval
                array[j]=temp
        return array
